def Ejercicio02(a):
    n = len(a)
    for i in range(n):
        swapped = False
        for j in range(1, n - i):
            if a[j] < a[j - 1]:
                a[j], a[j - 1] = a[j - 1], a[j]
                swapped = True
        if not swapped:
            break

# Ejemplo Clase
a = [3, 2, 5, 8, 4, 1]
Ejercicio02(a)
print(a)

print("Arreglo ordenado:", a)

[1, 2, 3, 4, 5, 8]
Arreglo ordenado: [1, 2, 3, 4, 5, 8]

def Ejercicio03(n):
    if n == 0:
        return 0
    else:
        x = 0
        y = 1
        for i in range(1, n):
            z = x + y
            x = y
            y = z
        return y

import matplotlib.pyplot as plt

def Ejercicio03(n):
    if n == 0:
        return 0
    else:
        x = 0
        y = 1
        for i in range(1, n):
            z = x + y
            x = y
            y = z
        return y

n = 11

# Generar lista de valores fib(1), fib(2), ..., fib(n)
fib = [Ejercicio03(i) for i in range(1, n + 1)]

# Calcular cocientes fib(n)/fib(n-1) desde n=2
phi_aprox = [fib[i] / fib[i - 1] for i in range(1, len(fib))]

# Mostrar tabla
print("| n | fib(n)/fib(n-1) |")
print("|---|------------------|")
for i in range(1, len(fib)):
    print(f"| {i+1} | {fib[i]}/{fib[i-1]} = {phi_aprox[i-1]:.6f} |")

# Graficar Fibonacci
plt.figure(figsize=(10,5))

plt.subplot(1,2,1)
plt.plot(range(1, n+1), fib, marker='o')
plt.title("Serie de Fibonacci")
plt.xlabel("n")
plt.ylabel("fib(n)")
plt.grid(True)

# Graficar cociente fib(n)/fib(n-1)
plt.subplot(1,2,2)
plt.plot(range(2, n+1), phi_aprox, marker='o', color='orange', label='fib(n)/fib(n-1)')
plt.axhline(y=(1 + 5 ** 0.5) / 2, color='red', linestyle='--', label='Número áureo φ')
plt.title("Aproximación al número áureo (φ)")
plt.xlabel("n")
plt.ylabel("Cociente")
plt.legend()
plt.grid(True)

plt.tight_layout()
plt.show()
